Plan Manager
Create, edit and delete membership plans. Changes are saved to Supabase and reflected live across the entire website.
⚠️ First-time setup: The membership_plans table does not exist yet in your Supabase project. 
-- Run this in Supabase SQL Editor → New Query
CREATE TABLE IF NOT EXISTS membership_plans (
  id            TEXT PRIMARY KEY DEFAULT gen_random_uuid()::text,
  plan_key      TEXT NOT NULL UNIQUE,
  label         TEXT NOT NULL,
  best_for      TEXT,
  days          INT DEFAULT 365,
  price_original NUMERIC DEFAULT 0,
  price         NUMERIC NOT NULL,
  discount      NUMERIC DEFAULT 0.10,
  savings       NUMERIC DEFAULT 0,
  indian_leads  INT DEFAULT 0,
  intl_leads    INT DEFAULT 0,
  lead_delivery TEXT DEFAULT 'WhatsApp & Email',
  website       BOOLEAN DEFAULT true,
  catalogue     INT DEFAULT 0,
  rm            BOOLEAN DEFAULT true,
  promotion     TEXT DEFAULT 'None',
  perf_report   BOOLEAN DEFAULT false,
  seo           BOOLEAN DEFAULT false,
  smo           BOOLEAN DEFAULT false,
  gst_support   TEXT DEFAULT 'Assistance',
  virtual_cards INT DEFAULT 1,
  email_support BOOLEAN DEFAULT true,
  iec           BOOLEAN DEFAULT true,
  msme          BOOLEAN DEFAULT true,
  fassai        TEXT DEFAULT 'Assistance',
  strategy_call BOOLEAN DEFAULT false,
  cta           TEXT DEFAULT 'Get Started',
  is_popular    BOOLEAN DEFAULT false,
  is_enterprise BOOLEAN DEFAULT false,
  color         TEXT DEFAULT '#1B3A5C',
  sort_order    INT DEFAULT 99,
  is_active     BOOLEAN DEFAULT true,
  created_at    TIMESTAMPTZ DEFAULT NOW(),
  updated_at    TIMESTAMPTZ DEFAULT NOW()
);
-- Enable Row Level Security but allow anon read
ALTER TABLE membership_plans ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_read_plans" ON membership_plans FOR SELECT TO anon USING (true);
CREATE POLICY "auth_all_plans"  ON membership_plans FOR ALL  TO authenticated USING (true);
